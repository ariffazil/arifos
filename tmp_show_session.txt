"""
core/state/session_manager.py — Constitutional Session Authority

Manages the lifecycle of GovernanceKernel instances (Ψ), ensuring isolation
and resource sovereignty across multiple concurrent intelligence sessions.
"""

import threading
import uuid
import sqlite3
import json
import os
from dataclasses import dataclass, field
from datetime import datetime
from typing import Any

from arifosmcp.core.governance_kernel import GovernanceKernel, AuthorityLevel, GovernanceState
from arifosmcp.core.shared.types import ActorIdentity


@dataclass
class SessionMetadata:
    """Active session telemetry and constraints."""

    session_id: str
    owner: str
    actor_identity: ActorIdentity | None = None
    created_at: datetime = field(default_factory=datetime.now)
    last_activity: datetime = field(default_factory=datetime.now)
    kernel: GovernanceKernel = field(init=False)


class SessionManager:
    """
    The Central State Authority for arifOS.
    Governs session ignition, isolation, and termination.
    Now hardened with SQLite persistence.
    """

    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        with cls._lock:
            if cls._instance is None:
                cls._instance = super().__new__(cls)
                cls._instance._sessions = {}
                cls._instance._db_path = os.path.join(os.getcwd(), "session_governance.db")
                cls._instance._init_db()
                cls._instance._load_sessions()
            return cls._instance

    def _init_db(self):
        """Initialize the persistence layer."""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute(
                """
                CREATE TABLE IF NOT EXISTS sessions (
                    session_id TEXT PRIMARY KEY,
                    owner TEXT,
                    actor_identity TEXT,
                    created_at TEXT,
                    last_activity TEXT,
                    kernel_state TEXT
                )
            """
            )

    def _load_sessions(self):
        """Restore sessions from the vault."""
        with sqlite3.connect(self._db_path) as conn:
            cursor = conn.execute("SELECT * FROM sessions")
            for row in cursor:
                sid, owner, actor_json, created_at, last_activity, kernel_json = row
                
                actor_identity = None
                if actor_json:
                    data = json.loads(actor_json)
                    actor_identity = ActorIdentity(**data)

                metadata = SessionMetadata(
                    session_id=sid, 
                    owner=owner, 
                    actor_identity=actor_identity,
                    created_at=datetime.fromisoformat(created_at),
                    last_activity=datetime.fromisoformat(last_activity)
                )

                # Re-hydrate the Kernel
                # Note: We create a fresh kernel and then update it with persisted state
                kernel = GovernanceKernel(decision_owner=owner, session_id=sid)
                if kernel_json:
                    kstate = json.loads(kernel_json)
                    # Mapping known fields back to kernel
                    for key, val in kstate.items():
                        # Skip read-only properties and methods to prevent overwriting logic with data
                        if key in ("thresholds", "environment", "energy", "void", "state_field", "temporal_stability") or callable(getattr(kernel, key, None)):
                            continue
                        if hasattr(kernel, key):
                            if key == "authority_level":
                                setattr(kernel, key, AuthorityLevel(val))
                            elif key == "governance_state":
                                setattr(kernel, key, GovernanceState(val))
                            else:
                                setattr(kernel, key, val)
                
                metadata.kernel = kernel
                self._sessions[sid] = metadata

    def _persist_session(self, sid: str):
        """Commit session state to the durable store."""
        metadata = self._sessions.get(sid)
        if not metadata:
            return

        actor_json = json.dumps(metadata.actor_identity.__dict__) if metadata.actor_identity else None
        kernel_json = json.dumps(metadata.kernel.to_dict())

        with sqlite3.connect(self._db_path) as conn:
            conn.execute(
                """
                INSERT OR REPLACE INTO sessions 
                (session_id, owner, actor_identity, created_at, last_activity, kernel_state)
                VALUES (?, ?, ?, ?, ?, ?)
                """,
                (
                    sid,
                    metadata.owner,
                    actor_json,
                    metadata.created_at.isoformat(),
                    metadata.last_activity.isoformat(),
                    kernel_json
                )
            )

    def create_session(
        self,
        owner: str,
        actor_identity: ActorIdentity | None = None,
        session_id: str | None = None,
    ) -> str:
        """
        Ignite a new governance session.
        F11: Establishes a unique authority boundary.
        """
        if not session_id:
            session_id = str(uuid.uuid4())

        metadata = SessionMetadata(
            session_id=session_id, owner=owner, actor_identity=actor_identity
        )

        # Initialize the Kernel for this session
        kernel = GovernanceKernel(decision_owner=owner, session_id=session_id)
        metadata.kernel = kernel

        self._sessions[session_id] = metadata
        self._persist_session(session_id)
        return session_id

    def get_session(self, session_id: str) -> SessionMetadata | None:
        """Retrieve the full session metadata for a specific session."""
        metadata = self._sessions.get(session_id)
        if metadata:
            metadata.last_activity = datetime.now()
            self._persist_session(session_id)
            return metadata
        return None

    def get_kernel(self, session_id: str) -> GovernanceKernel | None:
        """Retrieve the Ψ state for a specific session."""
        metadata = self.get_session(session_id)
        return metadata.kernel if metadata else None

    def get_identity(self, session_id: str) -> ActorIdentity | None:
        """Retrieve the ActorIdentity for a specific session."""
        metadata = self._sessions.get(session_id)
        return metadata.actor_identity if metadata else None

    def terminate_session(self, session_id: str):
        """Standard termination."""
        if session_id in self._sessions:
            del self._sessions[session_id]
            with sqlite3.connect(self._db_path) as conn:
                conn.execute("DELETE FROM sessions WHERE session_id = ?", (session_id,))

    def list_active_sessions(self) -> dict[str, str]:
        """Returns a map of session_id to owner."""
        return {sid: meta.owner for sid, meta in self._sessions.items()}


# Global singleton
session_manager = SessionManager()
